    unique = yes
	potential = {
		has_or_had_tag = FRA
		government_type = government_type:monarchy
		NOT = { has_reform = government_reform:feudal_nobility }
		NOT = { has_reform = government_reform:french_centralized_monarchy }
		NOT = { has_variable = enable_centralized_fra_reform }
	}
	allow = {
		has_or_had_tag = FRA
		NOT = { has_reform = government_reform:french_centralized_monarchy }
		NOT = { has_variable = enable_centralized_fra_reform }
	}
	locked = {
		custom_tooltip = {
			text = fra_removed_when_no_appanages_tt
			is_locked_mechanic = {
				mechanic = government_reform
				type = french_feudal_nobility
			}
		}
	}
	country_modifier = {
		has_appanages_subjects = yes
		diplomatic_capacity = 1
		cultures_capacity = 1	#UU Decrease from 2
		monthly_towards_serfdom = societal_value_large_monthly_move
		monthly_towards_aristocracy = societal_value_monthly_move
	}
	years = 2
}
